335bool Malla3D::RayIntersectsTriangle(glm::vec3 rayOrigin,
336 glm::vec3 rayVector,
337 glm::vec3 vertex0,
338 glm::vec3 vertex1,
339 glm::vec3 vertex2,
340 glm::vec3& outIntersectionPoint){
341 const float EPSILON = 0.00000001;
342 glm::vec3 edge1, edge2, h, s, q;
343 float a,f,u,v;
344 edge1 = vertex1 - vertex0;
345 edge2 = vertex2 - vertex0;
346 h = glm::cross(rayVector,edge2);
347 a = glm::dot(edge1,h);
348
349 if (a > -EPSILON && a < EPSILON){
350 return false;
351 }
352
353 f = 1.0/a;
354 s = rayOrigin - vertex0;
355 u = f * glm::dot(s,h);
356
357 if (u < 0.0 || u > 1.0){
358 return false;
359 }
360
361 q = glm::cross(s,edge1);
362 v = f * glm::dot(rayVector,q);
363
364 if (v < 0.0 || u + v > 1.0){
365 return false;
366 }
367
368 float t = f * glm::dot(edge2,q);
369
370 if (t > EPSILON){
371 outIntersectionPoint = rayOrigin + rayVector * t;
372 return true;
373 }
374 else{
375 return false;
376 }

675 glm::vec3 direction, std::vector<glm::vec3> &colisiones,
676 std::vector<int> &faces_hit){

683
684 glm::vec3 ray;
685 glm::vec3 normal;
686

707 ray = glm::normalize(ray);
708 normal = normals[faces_hit[ind_max]];

902std::vector<float> Malla3D::compute_panorama_symetry(){
903 float height_sym = panorama.size();
904 float width_sym = panorama[0].size();
905 float m = 0.4 * height_sym;
906 float n = 0.1 * width_sym;
907
908 float sym = 0;
909 float sym_diff = 0;
910 int index_up, index_down;
911 float up, down;
912
913 std::vector<float> sumas;
914
915 for(int w = 0; w < width_sym; w++){
916 sym = 0;
917 for(int h = (height_sym/2 - m); h <= (height_sym/2 + m); h++){
918 sym_diff = 0;
919 for(int x = 1; x <= n; x++){
920 index_up = w+x;
921 index_down = w-x;
922 if(index_down < 0){
923 index_down = (int)(width_sym + index_down);
924 }
925 if(index_up > (int)width_sym){
926 index_up = index_up % (int)width_sym;
927 }
928
929 up = panorama[h][index_up] / 255.0;
930 down = panorama[h][index_down] / 255.0;
931 sym_diff += abs(down - up);
932 }
933 sym_diff = sym_diff / n;
934 sym += sym_diff;
935 }
936
937 sym = sym / (2*m);
938 sym = 1.0 - sym;
939 sumas.push_back(sym);
940 }
941
942 return sumas;
943}
