261bool Malla3D::RayIntersectsTriangle(glm::vec3 rayOrigin,
262 glm::vec3 rayVector,
263 glm::vec3 vertex0,
264 glm::vec3 vertex1,
265 glm::vec3 vertex2,
266 glm::vec3& outIntersectionPoint){
267 const float EPSILON = 0.00000001;
268 glm::vec3 edge1, edge2, h, s, q;
269 float a,f,u,v;
270 edge1 = vertex1 - vertex0;
271 edge2 = vertex2 - vertex0;
272 h = glm::cross(rayVector,edge2);
273 a = glm::dot(edge1,h);
274
275 if (a > -EPSILON && a < EPSILON){
276 return false;
277 }
278
279 f = 1.0/a;
280 s = rayOrigin - vertex0;
281 u = f * glm::dot(s,h);
282
283 if (u < 0.0 || u > 1.0){
284 return false;
285 }
286
287 q = glm::cross(s,edge1);
288 v = f * glm::dot(rayVector,q);
289
290 if (v < 0.0 || u + v > 1.0){
291 return false;
292 }
293
294 float t = f * glm::dot(edge2,q);
295
296 if (t > EPSILON){
297 outIntersectionPoint = rayOrigin + rayVector * t;
298 return true;
299 }
300 else{
301 return false;
302 }

859float Malla3D::compute_panorama_symetry(){
860 float height_sym = panorama.size();
861 float width_sym = panorama[0].size();
862 float m = 0.4 * height_sym;
863 float n = 0.1 * width_sym;
864
865 float sym = 0;
866 float sym_diff = 0;
867 int index_up, index_down;
868 float up, down;
869
870 std::vector<float> sumas;
871
872 for(int w = 0; w < width_sym; w++){
873 sym = 0;
874 for(int h = (height_sym/2 - m); h <= (height_sym/2 + m); h++){
875 sym_diff = 0;
876 for(int x = 1; x <= n; x++){
877 index_up = w+x;
878 index_down = w-x;
879 if(index_down < 0){
880 index_down = (int)(width_sym + index_down);
881 }
882 if(index_up > (int)width_sym){
883 index_up = index_up % (int)width_sym;
884 }
885
886 up = panorama[h][index_up] / 255.0;
887 down = panorama[h][index_down] / 255.0;
888 sym_diff += abs(down - up);
889 }
890 sym_diff = sym_diff / n;
891 sym += sym_diff;
892 }
893
894 sym = sym / (2*m);
895 sym = 1.0 - sym;
896 sumas.push_back(sym);
897 }
898
899 float max = -1000000.0;
900 for(int i = 0; i < sumas.size(); i++){
901 if(max < sumas[i]){
902 max = sumas[i];
903 }
904 }
905 return max;
906}
