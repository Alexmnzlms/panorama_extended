290bool Malla3D::RayIntersectsTriangle(glm::vec3 rayOrigin,
291 glm::vec3 rayVector,
292 glm::vec3 vertex0,
293 glm::vec3 vertex1,
294 glm::vec3 vertex2,
295 glm::vec3& outIntersectionPoint){
296 const float EPSILON = 0.00000001;
297 glm::vec3 edge1, edge2, h, s, q;
298 float a,f,u,v;
299 edge1 = vertex1 - vertex0;
300 edge2 = vertex2 - vertex0;
301 h = glm::cross(rayVector,edge2);
302 a = glm::dot(edge1,h);
303
304 if (a > -EPSILON && a < EPSILON){
305 return false;
306 }
307
308 f = 1.0/a;
309 s = rayOrigin - vertex0;
310 u = f * glm::dot(s,h);
311
312 if (u < 0.0 || u > 1.0){
313 return false;
314 }
315
316 q = glm::cross(s,edge1);
317 v = f * glm::dot(rayVector,q);
318
319 if (v < 0.0 || u + v > 1.0){
320 return false;
321 }
322
323 float t = f * glm::dot(edge2,q);
324
325 if (t > EPSILON){
326 outIntersectionPoint = rayOrigin + rayVector * t;
327 return true;
328 }
329 else{
330 return false;
331 }

857float Malla3D::compute_panorama_symetry(){
858 float height_sym = panorama.size();
859 float width_sym = panorama[0].size();
860 float m = 0.4 * height_sym;
861 float n = 0.1 * width_sym;
862
863 float sym = 0;
864 float sym_diff = 0;
865 int index_up, index_down;
866 float up, down;
867
868 std::vector<float> sumas;
869
870 for(int w = 0; w < width_sym; w++){
871 sym = 0;
872 for(int h = (height_sym/2 - m); h <= (height_sym/2 + m); h++){
873 sym_diff = 0;
874 for(int x = 1; x <= n; x++){
875 index_up = w+x;
876 index_down = w-x;
877 if(index_down < 0){
878 index_down = (int)(width_sym + index_down);
879 }
880 if(index_up > (int)width_sym){
881 index_up = index_up % (int)width_sym;
882 }
883
884 up = panorama[h][index_up] / 255.0;
885 down = panorama[h][index_down] / 255.0;
886 sym_diff += abs(down - up);
887 }
888 sym_diff = sym_diff / n;
889 sym += sym_diff;
890 }
891
892 sym = sym / (2*m);
893 sym = 1.0 - sym;
894 sumas.push_back(sym);
895 }
896
897 float max = -1000000.0;
898 for(int i = 0; i < sumas.size(); i++){
899 if(max < sumas[i]){
900 max = sumas[i];
901 }
902 }
903 return max;
904}
